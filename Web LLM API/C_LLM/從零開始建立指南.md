### 2.2 建立基礎 API 伺服器 (api.py)

**第一版:最小可運行版本**

```python
import os
from dotenv import load_dotenv
import google.generativeai as genai
from flask import Flask, request, jsonify
from flask_cors import CORS

# 載入環境變數
load_dotenv('key.env')
API_KEY = os.getenv("GEMINI_API_KEY")

# 檢查 API 金鑰
if not API_KEY:
    print("❌ 錯誤:找不到 API 金鑰!")
    exit()

# 設定 Gemini
genai.configure(api_key=API_KEY)
model = genai.GenerativeModel('gemini-2.0-flash-exp')

# 建立 Flask 應用
app = Flask(__name__)
CORS(app)  # 允許跨域請求

# API 路由
@app.route('/api/chat', methods=['POST'])
def chat():
    try:
        # 接收使用者訊息
        data = request.json
        user_message = data.get('message', '')
        
        # 發送給 AI
        response = model.generate_content(user_message)
        
        # 返回結果
        return jsonify({
            'success': True,
            'response': response.text
        })
    except Exception as e:
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500

# 測試端點
@app.route('/api/test', methods=['GET'])
def test():
    return jsonify({'success': True, 'message': 'API 運作正常!'})

# 啟動伺服器
if __name__ == "__main__":
    print("🚀 API 伺服器啟動中...")
    print("📍 端點: http://localhost:5000")
    app.run(debug=True, host='0.0.0.0', port=5000)
```

**測試 API**:
```bash
python api.py
```

應該看到:
```
🚀 API 伺服器啟動中...
📍 端點: http://localhost:5000
 * Running on http://127.0.0.1:5000
```

### 2.4 加入日誌記錄功能

**在 api.py 中加入**:

```python
import json
from datetime import datetime
import uuid

# 日誌設定
LOGS_DIR = "logs"
LOGS_FILE = os.path.join(LOGS_DIR, "history.json")

# 確保目錄存在
if not os.path.exists(LOGS_DIR):
    os.makedirs(LOGS_DIR)

def save_log(user_message, ai_response, status="success"):
    """保存對話記錄"""
    log_entry = {
        "id": str(uuid.uuid4()),
        "timestamp": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
        "user_message": user_message,
        "ai_response": ai_response,
        "status": status
    }
    
    # 讀取現有日誌
    if os.path.exists(LOGS_FILE):
        with open(LOGS_FILE, 'r', encoding='utf-8') as f:
            logs = json.load(f)
    else:
        logs = []
    
    # 添加新記錄
    logs.append(log_entry)
    
    # 寫入文件
    with open(LOGS_FILE, 'w', encoding='utf-8') as f:
        json.dump(logs, f, ensure_ascii=False, indent=2)

# 在 chat() 函數中調用
@app.route('/api/chat', methods=['POST'])
def chat():
    try:
        data = request.json
        user_message = data.get('message', '')
        
        response = model.generate_content(user_message)
        ai_response = response.text
        
        # 記錄日誌
        save_log(user_message, ai_response, "success")
        
        return jsonify({
            'success': True,
            'response': ai_response
        })
    except Exception as e:
        save_log(user_message, None, "error")
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500
```

## 步驟三:前端介面開發

### 3.1 建立前端伺服器 (frontend_server.py)

```python
import http.server
import socketserver
import os

PORT = 8080

# 切換到當前目錄
os.chdir(os.path.dirname(os.path.abspath(__file__)))

# 建立伺服器
Handler = http.server.SimpleHTTPRequestHandler

with socketserver.TCPServer(("", PORT), Handler) as httpd:
    print(f"🌐 前端伺服器已啟動!")
    print(f"📍 請在瀏覽器開啟: http://localhost:{PORT}")
    print(f"按 Ctrl+C 停止伺服器\n")
    
    try:
        httpd.serve_forever()
    except KeyboardInterrupt:
        print("\n👋 伺服器已停止")
```

```html
        <h1>🤖 我的 LLM 應用</h1>
        
        <div class="chat-box" id="chatBox"></div>
        
        <div class="input-group">
            <input type="text" id="messageInput" placeholder="輸入訊息...">
            <button onclick="sendMessage()">發送</button>
        </div>

    <script>
        const API_URL = 'http://localhost:5000/api/chat';

        async function sendMessage() {
            const input = document.getElementById('messageInput');
            const message = input.value.trim();
            
            if (!message) return;
            
            // 顯示使用者訊息
            addMessage(message, 'user');
            input.value = '';
            
            try {
                // 發送到 API
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ message: message })
                });
                
                const result = await response.json();
                
                if (result.success) {
                    // 顯示 AI 回應
                    addMessage(result.response, 'ai');
                } else {
                    addMessage('錯誤: ' + result.error, 'ai');
                }
            } catch (error) {
                addMessage('連線失敗: ' + error.message, 'ai');
            }
        }

        function addMessage(text, type) {
            const chatBox = document.getElementById('chatBox');
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${type}-message`;
            messageDiv.textContent = text;
            chatBox.appendChild(messageDiv);
            chatBox.scrollTop = chatBox.scrollHeight;
        }

        // 按 Enter 發送
        document.getElementById('messageInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                sendMessage();
            }
        });
    </script>
</body>
</html>
```

## 常見問題

### Q1: 如何更換 AI 模型?

**A**: 修改 `api.py` 中的模型名稱:
```python
# 原本
model = genai.GenerativeModel('gemini-2.0-flash-exp')

# 更換為其他模型
model = genai.GenerativeModel('gemini-pro')
```

---

### Q2: 如何自訂 Prompt?

**A**: 在發送給 AI 前加入系統提示:
```python
def chat():
    user_message = data.get('message', '')
    
    # 自訂 prompt
    full_prompt = f"""
    你是一個友善的助手。
    請用繁體中文回答以下問題:
    
    {user_message}
    """
    
    response = model.generate_content(full_prompt)
```
